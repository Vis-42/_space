	\sss{(f)}{
	The time-independent Schrödinger equation for a free particle is:
	$$
		-\frac{\hbar^2}{2m} \pdv[2]{u_E(x)}{x} = E u_E(x)
		\imps \pdv[2]{u_E(x)}{x} + \frac{2mE}{\hbar^2} u_E(x) = 0
		\imps \pdv[2]{u_E(x)}{x} + k^2 u_E(x) = 0
		\quad \quad \lt[\because k = \sqrt{\frac{2mE}{\hbar^2}} \rt]
	$$
	For which the general solution is:
	\begin{equation*}
		u_E(x) = A e^{ikx} + B e^{-ikx}.
	\end{equation*}
	For a free particle, the potential is constant (or zero), so the system is invariant under translations. This implies that the energy eigenfunctions must also be translationally invariant. Thus, we choose plane waves as the eigenfunctions:
	$$
		u_E(x) = \frac{1}{\sqrt{2\pi}} e^{ikx}\quad \text{ ,where $k = \sqrt{\frac{2mE}{\hbar^2}}$}
	$$
	Since $k = \sqrt{\dfrac{2mE}{\hbar^2}}$, the energy eigenvalues are:
	\begin{equation*}
		E = \frac{\hbar^2 k^2}{2m}.
	\end{equation*}
	Note that $E \geq 0$ because the kinetic energy of a free particle cannot be negative.

		&= \sqrt{\frac{\sigma_0}{\sqrt{2}\pi^{3/2}\hbar^2}}\qmi \e{\frac{ix_0}{\hbar}(p_0 - p)-\frac{\sigma_0^2}{\hbar^2} (p_0 - p)^2 + \frac{ipx}{\hbar} -\frac{ip^2 t}{2m\hbar}}\dd p \\
	\end{align*}
	Here,
	\begin{align*}
		&\quad \frac{ix_0}{\hbar}(p_0 - p)-\frac{\sigma_0^2}{\hbar^2} (p_0 - p)^2 + \frac{ipx}{\hbar} -\frac{ip^2 t}{2m\hbar}\\
		&= -p^2\qty(\frac{\sigma_0^2}{\hbar^2} + \frac{it}{2m\hbar}) +p\qty(\frac{2\sigma_0^2p_0}{\hbar^2} + \frac{i(x-x_0)}{\hbar}) -\qty(\frac{p_0^2\sigma_0^2}{\hbar^2} - \frac{ix_0p_0}{\hbar})\\
		&= -Ap^2 +Bp-C\\
		&= -A\lt[p-\frac{B}{2A}\rt]^2 +\frac{B^2}{4A} -C
	\end{align*}
	$$\text{where, } A=\qty(\frac{\sigma_0^2}{\hbar^2} + \frac{it}{2m\hbar}) \qc B=\qty(\frac{2\sigma_0^2p_0}{\hbar^2} + \frac{i(x-x_0)}{\hbar}) \qc C=\qty(\frac{p_0^2\sigma_0^2}{\hbar^2} - \frac{ix_0p_0}{\hbar})$$
	Consequently,
	\begin{align*}
		\psi(x, t) &=\sqrt{\frac{\sigma_0}{\sqrt{2}\pi^{3/2}\hbar^2}} \qmi \e{-A\lt[p-\frac{B}{2A}\rt]^2 +\frac{B^2}{4A} -C}\dd p\\
		&=\sqrt{\frac{\sigma_0}{\sqrt{2}\pi^{3/2}\hbar^2}}
		\e{\frac{B^2}{4A} -C} \qmi \e{-A\lt[p-\frac{B}{2A}\rt]^2}\dd p\\
		&=\sqrt{\frac{\sigma_0}{\sqrt{2}\pi^{3/2}\hbar^2}}
		\e{\frac{B^2}{4A} -C}\sqrt{\frac{\pi}{A}} && \lt[\because \qmi \e{-a y^2} \dd{y} = \sqrt{\frac{\pi}{a}}\rt]\\
		&=\sqrt{\frac{\sigma_0}{\sqrt{2}\pi^{1/2}\hbar^2A}}
		\e{\frac{B^2}{4A} -C}\\
		&=\sqrt{\frac{\sigma_0}{\sqrt{2\pi}\hbar^2\qty(\frac{\sigma_0^2}{\hbar^2} + \frac{it}{2m\hbar})}}\e{\frac{B^2}{4A} -C}\\
		&=\frac{1}{\sqrt{\sqrt{2\pi}\sigma_0\qty(1 + \frac{i\hbar t}{2m\sigma_0^2})}}\e{\frac{B^2}{4A} -C} \tag{5} \label{05}\\
	\end{align*}
	Here,
	\begin{align*}
		\frac{B^2}{4A} -C &= \frac{\qty(\frac{2\sigma_0^2p_0}{\hbar^2} + \frac{i(x-x_0)}{\hbar})^2}{4\qty(\frac{\sigma_0^2}{\hbar^2} + \frac{it}{2m\hbar})} -\qty(\frac{p_0^2\sigma_0^2}{\hbar^2} - \frac{ix_0p_0}{\hbar})\\
		&=\frac{\hbar^2\qty(\frac{2\sigma_0^2p_0}{\hbar^2} + \frac{i(x-x_0)}{\hbar})^2}{4\sigma_0^2\qty(1 + \frac{i\hbar t}{2m\sigma_0^2})} -\frac{p_0^2\sigma_0^2}{\hbar^2}+ \frac{ix_0p_0}{\hbar}\\
		&=\frac{\qty(\frac{4\sigma_0^4p_0^2}{\hbar^2} -(x-x_0)^2+\frac{i}{\hbar}4\sigma_0^2p_0(x-x_0))}{4\sigma_0^2\qty(1 + \frac{i\hbar t}{2m\sigma_0^2})} -\frac{p^2\sigma_0^2}{\hbar^2}+ \frac{ix_0p_0}{\hbar}\\
		&=\frac{\qty(\frac{4\sigma_0^4p_0^2}{\hbar^2})}{4\sigma_0^2\qty(1 + \frac{i\hbar t}{2m\sigma_0^2})} + \frac{\qty(-(x-x_0-\frac{p_0}{m}t)^2)+\frac{p_0^2}{m^2}t^2 -\frac{2p_0(x-x_0)}{m}t}{4\sigma_0^2\qty(1 + \frac{i\hbar t}{2m\sigma_0^2})} + \frac{\qty(\frac{i}{\hbar}4\sigma_0^2p_0(x-x_0))}{4\sigma_0^2\qty(1 + \frac{i\hbar t}{2m\sigma_0^2})} -\frac{p_0^2\sigma_0^2}{\hbar^2}+ \frac{ix_0p_0}{\hbar}\\
		&= \frac{\sigma_0^2 p_0^2}{\hbar^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} - \frac{p_0^2 \sigma_0^2}{\hbar^2} - \frac{(x - x_0 - \frac{p_0}{m} t)^2}{4 \sigma_0^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{\frac{p_0^2}{m^2} t^2 - \frac{2 p_0 (x - x_0)}{m} t}{4 \sigma_0^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{i p_0 (x - x_0)}{\hbar \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{i x_0 p_0}{\hbar} \\
		&= - \frac{p_0^2 \sigma_0^2}{\hbar^2}\qty(1 - \frac{1}{1 + \frac{i \hbar t}{2 m \sigma_0^2}}) - \frac{(x - x_0 - \frac{p_0}{m} t)^2}{4 \sigma_0^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{p_0^2 t^2 - 2 p_0 (x - x_0)m t}{4 \sigma_0^2m^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{i p_0 x}{\hbar \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{i x_0 p_0}{\hbar} \qty(1 - \frac{1}{1 + \frac{i \hbar t}{2 m \sigma_0^2}}) \\
		&= \frac{-i p_0^2t}{2 m \hbar \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} - \frac{(x - x_0 - \frac{p_0}{m} t)^2}{4 \sigma_0^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{p_0^2 t^2 - 2 p_0 (x - x_0)m t}{4 \sigma_0^2m^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{i p_0 x}{\hbar \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} - \frac{x_0 p_0 t}{2 m \sigma_0^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} \\
		&= - \frac{(x - x_0 - \frac{p_0}{m} t)^2}{4 \sigma_0^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} +\frac{-i p_0^2t}{2 m \hbar \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{p_0^2 t^2 - 2 p_0 x m t}{4 \sigma_0^2m^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{i p_0 x}{\hbar \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} \\
		&= - \frac{(x - x_0 - \frac{p_0}{m} t)^2}{4 \sigma_0^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} +\frac{-i p_0^2t}{2 m \hbar \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{p_0^2 t^2 - 2 p_0 x m t}{4 \sigma_0^2m^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{i p_0 x}{\hbar \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})} + \frac{i}{\hbar}\qty(p_0x-\frac{p_0^2}{2m}t) - \frac{i}{\hbar}\qty(p_0x-\frac{p_0^2}{2m}t)\\
		&=- \frac{(x-x_0-\frac{p_0}{m}t)^2}{4\sigma_0^2\qty(1 + \frac{i\hbar t}{2m\sigma_0^2})}+ \frac{i}{\hbar}\qty(p_0x-\frac{p_0^2}{2m}t) +  \frac{p_0^2 t^2 - 2 p_0 x m t}{4 \sigma_0^2m^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})}- \frac{i}{\hbar}\qty(p_0x-\frac{p_0^2}{2m}t)\qty(1 - \frac{1}{1 + \frac{i \hbar t}{2 m \sigma_0^2}})\\
		&=- \frac{(x-x_0-\frac{p_0}{m}t)^2}{4\sigma_0^2\qty(1 + \frac{i\hbar t}{2m\sigma_0^2})}+ \frac{i}{\hbar}\qty(p_0x-\frac{p_0^2}{2m}t) +  \frac{p_0^2 t^2 - 2 p_0 x m t}{4 \sigma_0^2m^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})}- \frac{p_0^2 t^2 - 2 p_0 x m t}{4 \sigma_0^2m^2 \qty(1 + \frac{i \hbar t}{2 m \sigma_0^2})}\\
		&=- \frac{(x-x_0-\frac{p_0}{m}t)^2}{4\sigma_0^2\qty(1 + \frac{i\hbar t}{2m\sigma_0^2})}+ \frac{i}{\hbar}\qty(p_0x-\frac{p_0^2}{2m}t) \tag{6} \label{06}\\
	\end{align*}
	Using \eqref{05} and \eqref{06},
	$$\boxed{\psi(x, t)=\frac{1}{\sqrt{\sqrt{2\pi}\sigma_0\qty(1 + \dfrac{i\hbar t}{2m\sigma_0^2})}}\e{\dfrac{i}{\hbar}\qty(p_0x-\dfrac{p_0^2}{2m}t)}\e{- \dfrac{\qty(x-x_0-\dfrac{p_0}{m}t)^2}{4\sigma_0^2\qty(1 + \dfrac{i\hbar t}{2m\sigma_0^2})}}}$$
	\vspace{1em}
	}

	\wa
	\sss{(h)}{
	We have,
	$$
	\begin{cases}
		\psi(x, t)&=\frac{1}{\sqrt{\sqrt{2\pi}\sigma_0\qty(1 + \dfrac{i\hbar t}{2m\sigma_0^2})}}\e{\dfrac{i}{\hbar}\qty(p_0x-\dfrac{p_0^2}{2m}t)}\e{- \dfrac{\qty(x-x_0-\dfrac{p_0}{m}t)^2}{4\sigma_0^2\qty(1 + \dfrac{i\hbar t}{2m\sigma_0^2})}}\\
		\psi^{*}(x, t)&=\frac{1}{\sqrt{\sqrt{2\pi}\sigma_0\qty(1 - \dfrac{i\hbar t}{2m\sigma_0^2})}}\e{-\dfrac{i}{\hbar}\qty(p_0x-\dfrac{p_0^2}{2m}t)}\e{- \dfrac{\qty(x-x_0-\dfrac{p_0}{m}t)^2}{4\sigma_0^2\qty(1 - \dfrac{i\hbar t}{2m\sigma_0^2})}}\\
	\end{cases}
	$$
	$$\imps \rho(x,t)=|\psi(x,t)|^2= \frac{1}{\sqrt{2\pi\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\e{-\dfrac{\qty(x-x_0-\dfrac{p_0}{m}t)^2}{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}$$
	\paragraph{i.}
	\begin{align*}
		\ev{x}_t&=\qmi\psi^{*}(x,t)[x]\psi(x,t)\cdot \dd x\\
		&=\qmi x |\psi(x,t)|^2 \cdot\dd x\\
		&=\frac{1}{\sqrt{\pi}}\qmi\pqty{\frac{x-x_0-\dfrac{p_0}{m}t+x_0+\dfrac{p_0}{m}t}{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}}\e{-\dfrac{\qty(x-x_0-\dfrac{p_0}{m}t)^2}{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\cdot \dd x\\ \\
		&\quad \quad \bqty{\because y=\frac{x-x_0-\dfrac{p_0}{m}t}{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\Rightarrow \dd y=\frac{\dd x}{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}}\\ \\
		&=\frac{1}{\sqrt{\pi}}\bqty{\qmi y\exp\pqty{-y^{2}}\cdot{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\dd y + \frac{x_0+\dfrac{p_0}{m}t}{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\qmi\exp\pqty{-y^{2}}\cdot{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\dd y}\\
		&=\frac{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}{\sqrt{\pi}}\bqty{0 + \frac{x_0+\dfrac{p_0}{m}t}{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\sqrt{\pi}} \quad \quad \bqty{\because y\exp(-y^2) \text{is an odd function and} \atop \qmi\exp\pqty{-y^{2}}\cdot \dd y =\sqrt{\pi}} \\
		&\\
		\imps & \boxed{\ev{x}_t = x_0 + \frac{p_0}{m} t}
	\end{align*}

	\paragraph{ii.}
	We have,
	\begin{align*}
		\ev{x^2}_0&=\qmi\psi^{*}(x,t)[x]^2\psi(x,t)\cdot \dd x\\
		&=\qmi\qty[\qty(x-x_0-\dfrac{p_0}{m}t)^2 + 2x\qty(x_0+\dfrac{p_0}{m}t) -\qty(x_0+\dfrac{p_0}{m}t)^2] |\psi(x,t)|^2 \cdot\dd x\\
		&=\qty[\qmi\qty(x-x_0-\dfrac{p_0}{m}t)^2|\psi(x,t)|^2\dd x]+\qty[2\qty(x_0+\dfrac{p_0}{m}t)\qmi[x]|\psi(x,t)|^2\dd x]-\qty[\qty(x_0+\dfrac{p_0}{m}t)^2\qmi|\psi(x,0)|^2\dd x]\\
		&=\qty[\frac{1}{\sqrt{2\pi\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\qmi\qty(x-x_0-\dfrac{p_0}{m}t)^2\e{- \dfrac{\qty(x-x_0-\dfrac{p_0}{m}t)^2}{2\sigma_0^2\qty(1 + \dfrac{i\hbar t}{2m\sigma_0^2})}}\dd x]+2\qty(x_0+\dfrac{p_0}{m}t)\ev{x}_t-\qty(x_0+\dfrac{p_0}{m}t)^2\\ \\
		&\quad \quad \bqty{\because y=\frac{x-x_0-\dfrac{p_0}{m}t}{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\Rightarrow \dd y=\frac{\dd x}{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}}\\ \\
		&=\qty[\frac{1}{\sqrt{2\pi\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\qmi\qty(y{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}})^2\exp\pqty{-y^{2}}\cdot{\sqrt{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}}\dd y]+\qty(x_0+\dfrac{p_0}{m}t)^2\\
		&=\frac{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}{\sqrt{\pi}}\bqty{\qmi   y^2\exp\pqty{-y^{2}}\cdot\dd y}+\qty(x_0+\dfrac{p_0}{m}t)^2 \tag{7} \label{07}\\
	\end{align*}
	Here, with $a > 0$,
	\[\qmi \exp(-a x^2) \dd{x} = \frac{1}{\sqrt{a}} \qmi \exp(-a x^2) \dd{(\sqrt{a} x)} = \sqrt{\frac{\pi}{a}}\]
	It follows that
	\[\qmi x^2 \exp(-a x^2) \dd{x} = -\pdv{}{a} \qmi \exp(-a x^2) \dd{x} = \frac{1}{2} \sqrt{\frac{\pi}{a^3}}\]
	Using this is \eqref{07},
	\begin{align*}
		\ev{x^2}_0&=\frac{2\sigma_0^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)}{\sqrt{\pi}}\bqty{\frac{\sqrt{\pi}}{2}}+x_0^2\\
		\ev{x^2}_0-x_0^2&=\sigma_{0}^2\qty(1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2)\\
		\sqrt{\ev{x^2}_0-\ev{x}^2}&=\sigma_{0}\sqrt{1 + \qty(\dfrac{\hbar t}{2m\sigma_0^2})^2} \imps \boxed{\Delta x(t)=\sigma_0 \sqrt{1 + \lt(\frac{\hbar t}{2m\sigma_0^2}\rt)^2}}\\
	\end{align*}
	\paragraph{Verifying the expression for the expectation value of velocity,}
	$$\dv{t} \ev{x}_t = \dv{t} \qty(x_0 + \frac{p_0}{m} t) = \frac{p_0}{m} = \frac{1}{m} \ev{p}_t \imps \boxed{\dv{t} \ev{x}_t = \frac{1}{m} \ev{p}_t}$$
	}

	\wa
	\sss{(i)}{
	We have,
	\begin{align*}
		\hat{H} \phi_p(t) &= i\hbar \pdv{t} \phi_p(t) \imps
		\frac{p^2}{2m} \phi_p(t) = i\hbar \pdv{t} \phi_p(t) \imps
		\pdv{t} \phi_p(t) = \frac{-i}{\hbar} \frac{p^2}{2m} \phi_p(t) \\ \\
		&\imps
		\boxed{\phi_p(t) = \phi_p(0) \e{- \frac{i p^2 t}{2m\hbar}}} \\
	\end{align*}
	Note that $ \abs{\phi_p(t)}^2 = \abs{\phi_p(0)}^2 $, which implies that $ \abs{\phi_p(t)}^2 $ does not depent on time $t$.\\
	Hence,
	\begin{align*}
		\ev{p}_t &= \int_{-\infty}^{\infty} \phi_p^*(t) p \phi_p(t) \dd{p} = \qmi p \abs{\phi_p(t)}^2 \dd{p} = \qmi p \abs{\phi_p(0)}^2 \dd{p} = \ev{p}_0 = p_0
	\end{align*}
	Therefore, the independence of $ \abs{\phi_p(t)}^2 $ with respect to time $t$ implies that the expectation value of momemtum $\ev{p}_t$ is conserved for the particle.\\
	}
